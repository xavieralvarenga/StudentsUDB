private static void menu(Scanner input, Map<Integer, String> alumnos) {

    int opcion = 0;

    do {
        System.out.println("\n===== MENU UDB VIRTUAL =====");
        System.out.println("1. Ingresar alumno");
        System.out.println("2. Buscar alumno");
        System.out.println("3. Eliminar alumno");
        System.out.println("4. Mostrar alumnos");
        System.out.println("5. Salir");

        try {
            opcion = input.nextInt();
            input.nextLine(); // limpiar buffer

            switch (opcion) {
                case 1:
                    ingresarAlumno(input, alumnos);
                    break;
                case 2:
                    buscarAlumno(input, alumnos);
                    break;
                case 3:
                    eliminarAlumno(input, alumnos);
                    break;
                case 4:
                    mostrarAlumnos(alumnos);
                    break;
                case 5:
                    System.out.println("Saliendo del sistema...");
                    break;
                default:
                    System.out.println("Opción inválida.");
            }

        } catch (Exception e) {
            System.out.println("Error: Ingrese solo números.");
            input.nextLine(); // limpiar buffer
        }

    } while (opcion != 5);
}
